             throws NullPointerException {
         if (n1 == null) {
             throw new NullPointerException("n1 is null");
         }
         if (head == null) {
             head = n1; size++; return;
         }
         if (n2 == null) {
             throw new NullPointerException("n2 is null");
         }
         if (n2 == head) {
             n1.setNext(n2);
             head = n1; size++;
             return;
         }
         Node n = new Node();
         n.data = n2.getData();

     }
